git init    = serve para iniciar um repositório do git
git config user.name seunome    = serve para salvar os projetos que eu fizer no git com meu nome, serve para identificação
git config user.email seuemail     = seve para salvar os projetos que eu fizer com meu email, serve para identificação
git add . = adiciona um arquivo ao repositório
git commit -m= serve para salvar a versão do arquivo e serve para explicar o que foi modificado ou adicionado
